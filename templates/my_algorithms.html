def duplicate_string(my_string):

    my_dict = {}

    string_status = 'All Unique'

    for c in my_string:

        if c in my_dict:

            my_dict[c] = my_dict[c]+1

            string_status = 'Duplicate Found'

        else:

            my_dict[c] = 1

    return string_status

    while my_number % 2 == 0:

        prime_factors.append(2)
